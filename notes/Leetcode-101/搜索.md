## 695. 岛屿的最大面积

[695. 岛屿的最大面积](https://leetcode-cn.com/problems/max-area-of-island/)

```html
找到给定的二维数组中最大的岛屿面积。(如果没有岛屿，则返回面积为 0 。)
```

```cpp
class Solution {
public:
    int dir[4][2] = {0, 1, 1, 0, -1, 0, 0, -1};
    int n, m;
    void dfs(vector<vector<int>>& grid, int x, int y, int &sum) {
        sum++;
        for(int i = 0; i < 4; i++) {
            int dx = x + dir[i][0];
            int dy = y + dir[i][1];
            if(dx < 0 || dx >= n || dy < 0 || dy >= m )continue;
            if(grid[dx][dy] == 1) {
                grid[dx][dy] = 0;
                dfs(grid, dx, dy, sum);
            }
        }
        return;
    }
    int maxAreaOfIsland(vector<vector<int>>& grid) {
         n = grid.size();
         m = grid[0].size();
        int mmax = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                int sum = 0;
                if(grid[i][j] == 1) {
                    grid[i][j] = 0;
                    dfs(grid, i, j, sum);
                }
                mmax = max(sum, mmax);
            }
        }
        return mmax;
    }
};
```

## 417. 太平洋大西洋水流问题

[417. 太平洋大西洋水流问题](https://leetcode-cn.com/problems/pacific-atlantic-water-flow/)

从两片海域分别编号为1和2，然后往中间搜索，最后图中为3的为能流到两片海域的山地

```cpp
class Solution {
public:

    int dir[4][2] = {0, 1, 1, 0, -1, 0, 0, -1};
    bool valid(int r, int c, int R, int C) {
        return r >= 0 && r < R && c >= 0 && c < C;
    }
    vector<vector<int>> pacificAtlantic(vector<vector<int>>& matrix) {
        vector<vector<int>>res;
        if(matrix.empty()) return res;
        int R = matrix.size();
        int C = matrix[0].size();
        vector<vector<int> >status(R, vector<int>(C, 0));
        queue<pair<int, int>>q;
        for(int i = 0; i < R; i++) {
            q.push({i, 0});
            status[i][0] |= 1;//+1
            q.push({i, C - 1});
            status[i][C - 1] |= 2;//+2
        }
        for(int i = 0; i < C; i++) {
            q.push({0, i});
            status[0][i] |= 1;
            q.push({R - 1, i});
            status[R - 1][i] |= 2;
        }
        while(!q.empty()) {
            auto p = q.front();
            q.pop();
            for(int i = 0; i < 4; i++) {
                int r = p.first + dir[i][0];
                int c = p.second + dir[i][1];
                if(valid(r, c, R, C) && matrix[r][c] >= matrix[p.first][p.second]) {
                    if(status[r][c] != status[p.first][p.second]) {
                        status[r][c] |= status[p.first][p.second];//相遇 = 3
                        q.push({r, c});
                    }
                } 
            }
        }
        for(int i = 0; i < R; i++) {
            for(int j = 0; j < C; j++) {
                if(status[i][j] == 3) {
                    res.push_back({i, j});
                }
            }
        }
        return res;
    }
};
```

## 46. 全排列

[46. 全排列](https://leetcode-cn.com/problems/permutations/)

```cpp
class Solution {
public:
    vector<vector<int>>res;
    void dfs(vector<int>&now, vector<int>&nums, vector<bool>&check) {
        if(now.size() == nums.size()) {
            res.push_back(now);
            return;
        }
        for(int i = 0; i < nums.size(); i++) {
            if(!check[i]) {
                check[i] = true;
                now.push_back(nums[i]);
                dfs(now, nums, check);
                check[i] = false;
                now.pop_back();
            }
        }
        return;
    }
    vector<vector<int>> permute(vector<int>& nums) {
        vector<int>ans;
        vector<bool>check(nums.size(), false);
        dfs(ans, nums, check);
        return res;
    }
};
```

## 77. 组合

[77. 组合](https://leetcode-cn.com/problems/combinations/)

```cpp
class Solution {
public:
    vector<vector<int>> res;
    vector<int>ans;
    void dfs(int s, int n, int k) {
        if(ans.size() == k) {
            res.push_back(ans);
            return;
        }
        for(int i = s; i <= n; i++) {
            ans.push_back(i);
            dfs(i + 1, n, k);
            ans.pop_back();
        }
    }
    vector<vector<int>> combine(int n, int k) {
        dfs(1, n, k);
        return res;
    }
};
```
