## 695. 岛屿的最大面积

[695. 岛屿的最大面积](https://leetcode-cn.com/problems/max-area-of-island/)

```html
找到给定的二维数组中最大的岛屿面积。(如果没有岛屿，则返回面积为 0 。)
```

```cpp
class Solution {
public:
    int dir[4][2] = {0, 1, 1, 0, -1, 0, 0, -1};
    int n, m;
    void dfs(vector<vector<int>>& grid, int x, int y, int &sum) {
        sum++;
        for(int i = 0; i < 4; i++) {
            int dx = x + dir[i][0];
            int dy = y + dir[i][1];
            if(dx < 0 || dx >= n || dy < 0 || dy >= m )continue;
            if(grid[dx][dy] == 1) {
                grid[dx][dy] = 0;
                dfs(grid, dx, dy, sum);
            }
        }
        return;
    }
    int maxAreaOfIsland(vector<vector<int>>& grid) {
         n = grid.size();
         m = grid[0].size();
        int mmax = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                int sum = 0;
                if(grid[i][j] == 1) {
                    grid[i][j] = 0;
                    dfs(grid, i, j, sum);
                }
                mmax = max(sum, mmax);
            }
        }
        return mmax;
    }
};
```

   ```cpp
   class Solution {
   public:
       int find(int *pir, int x) {
           while(x != pir[x]) {
               x = pir[x];
           }
           return x;
       }
       int findCircleNum(vector<vector<int>>& isConnected) {
           int pir[200];
           int n = isConnected.size();
           int m = isConnected[0].size();
           for(int i = 0; i < n; i++) {
               pir[i] = i;
           }
           for(int i = 0; i < n; i++) {
               for(int j = 0; j < m; j++) {
                   if(isConnected[i][j] == 1) {
                       int a = find(pir, i);
                       int b = find(pir, j);
                       if(a != b) {
                           pir[a] = b;
                       }
                   }
               }
           }
           int sum = 0;
           for(int i = 0; i < n; i++) {
               if(pir[i] == i) sum++;
           }
           return sum;
       }
   };
   ```
