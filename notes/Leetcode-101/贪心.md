# 贪心

#### [455. 分发饼干](https://leetcode-cn.com/problems/assign-cookies/)

```cpp
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int i = 0, j = 0;
        while(i < g.size()&& j < s.size()) {
            if(g[i] <= s[j]) {//一直控制g[i] <= s[j], 同时往后走
                i++;
            }
            j++;//如果g[i] > s[j],j也得往后走啊
        }
        return i;
    }
};
```
